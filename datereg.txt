CREATE TRIGGER set_default_reg_date
BEFORE INSERT ON "Users"
FOR EACH ROW
WHEN NEW.reg_date IS NULL
BEGIN
  UPDATE "Users" SET reg_date = CURRENT_DATE WHERE user_ID = NEW.user_ID;
END;
